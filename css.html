font-size: xx-small 10px = xx-small, x-small 10px = x-small, small 12px = small, medium 16px = medium, large 18px = large, x-large 24px = x-large, xx-large 32px = xx-large
font-size: smaller 30px(article) * 5/6 = 25px = smaller larger 30px(article) * 6/5 = 36px = larger
font-size: 1em 1em = parents' size, 1rem 1rem = html's size(default 16 px), 1ex 19px(현재 폰트에서 19px가 소문자 x의 크기) = 1ex, 20px = (20 * 0.75)pt, 150% = (30 * 1.5)px
